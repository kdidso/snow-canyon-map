      function normalizeText(str) {
        return (str || "")
          .toLowerCase()
          .normalize("NFD")
          .replace(/[\u0300-\u036f]/g, "")
          .replace(/[^a-z0-9\s]/g, "")
          .replace(/\s+/g, " ")
          .trim();
      }

      function updateClearButton() {
        const inputEl = document.getElementById("search-query-input");
        const clearBtn = document.getElementById("clear-search-btn");
        clearBtn.style.display = inputEl.value.trim() ? "flex" : "none";
      }

function getFilteredPeople(query) {
  const q = normalizeText(query);

  if (q.length < 3) return [];

  return people.filter(person => {
    const name = normalizeText(person.name);
    return name.includes(q);
  }).slice(0, 8);
}

      function renderCards(list) {
        const resultsEl = document.getElementById("results");
        const emptyEl = document.getElementById("no-results");

        resultsEl.innerHTML = "";

        if (!list.length) {
          emptyEl.style.display = "block";
          return;
        }

        emptyEl.style.display = "none";

        list.forEach(person => {
          const card = document.createElement("a");
          card.className = "person-card";
          card.href = person.url;
          card.target = "_blank";
          card.rel = "noopener noreferrer";

          const imgWrap = document.createElement("div");
          imgWrap.className = "person-image-wrap";

          const img = document.createElement("img");
          img.className = "person-image";
          img.src = person.image;
          img.alt = person.name;
          img.loading = "lazy";

          img.onerror = function () {
            this.style.display = "none";
            imgWrap.classList.add("image-missing");
          };

          imgWrap.appendChild(img);

          const name = document.createElement("div");
          name.className = "person-name";
          name.textContent = person.name;

          card.appendChild(imgWrap);
          card.appendChild(name);

          resultsEl.appendChild(card);
        });
      }

function updateResults() {
  const inputEl = document.getElementById("search-query-input");
  const query = inputEl.value.trim();
  const filtered = getFilteredPeople(query);
  const noResultsEl = document.getElementById("no-results");

  renderCards(filtered);

  if (normalizeText(query).length < 3) {
    noResultsEl.style.display = "none";
  } else if (filtered.length === 0) {
    noResultsEl.textContent = "No matching names found.";
    noResultsEl.style.display = "block";
  } else {
    noResultsEl.style.display = "none";
  }

  updateClearButton();
}

function clearSearch() {
  const inputEl = document.getElementById("search-query-input");

  inputEl.value = "";
  updateClearButton();
  renderCards([]);
  document.getElementById("no-results").style.display = "none";
  inputEl.focus();
}

      function submitSearch() {
        const input = document.getElementById("search-query-input").value.trim();
        if (!input) return;

        const exact = people.find(
          person => normalizeText(person.name) === normalizeText(input)
        );

        if (exact) {
          window.open(exact.url, "_blank");
          return;
        }

        const partial = people.find(
          person => normalizeText(person.name).includes(normalizeText(input))
        );

        if (partial) {
          window.open(partial.url, "_blank");
          return;
        }

        alert("No matching name found!");
      }

      function searchKeyPress(e) {
        e = e || window.event;
        if (e.key === "Enter" || e.keyCode === 13) {
          submitSearch();
          return false;
        }
        return true;
      }

window.addEventListener("DOMContentLoaded", () => {
  const inputEl = document.getElementById("search-query-input");
  const clearBtn = document.getElementById("clear-search-btn");

  inputEl.addEventListener("input", updateResults);
  clearBtn.addEventListener("click", clearSearch);

  inputEl.addEventListener("keydown", e => {
    if (e.key === "Escape") {
      clearSearch();
    }
  });

  updateClearButton();
});
